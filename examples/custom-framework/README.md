# Custom Framework Example

This example demonstrates how to use the Wink View Generator package with your own custom CSS framework, showing how the package generates semantic HTML that can be styled with any CSS approach.

## Overview

We'll create a complete CRUD interface using:

- **Custom CSS Framework** - No Bootstrap or Tailwind dependencies
- **Semantic HTML** - Clean, framework-agnostic markup
- **BEM Methodology** - Organized CSS class naming
- **CSS Custom Properties** - Modern CSS variables for theming
- **CSS Grid & Flexbox** - Modern layout techniques
- **Dark Mode Support** - CSS custom properties for theme switching
- **Responsive Design** - Mobile-first approach

## Features Showcased

- ✅ **Framework-agnostic HTML** generated by Wink View Generator
- ✅ **Custom CSS architecture** using modern techniques
- ✅ **Component-based styling** with reusable patterns
- ✅ **Theme system** with light/dark mode support
- ✅ **Responsive design** without framework dependencies
- ✅ **Accessible markup** with proper ARIA labels
- ✅ **Performance optimized** CSS with minimal footprint

## Getting Started

### 1. Generate Views with Custom Framework

Use the `custom` framework option to generate semantic HTML:

```bash
# Generate CRUD views without framework-specific classes
php artisan wink:views:crud products --framework=custom --ajax --search --sorting

# Generate components
php artisan wink:views:components --framework=custom --all

# Generate layouts
php artisan wink:views:layouts --framework=custom --navigation --sidebar
```

            const html = document.documentElement;
            const currentTheme = html.getAttribute('data-theme');

            const icon = document.getElementById('theme-icon');

Add JavaScript for enhanced functionality:

```javascript
// public/js/app.js
document.addEventListener('DOMContentLoaded', function() {
    // Initialize functionality
    initializeSelectAll();
    initializeBulkActions();
    initializeSearch();
    initializeModals();
});

function initializeSelectAll() {
    const selectAllCheckbox = document.getElementById('select-all');
    const rowCheckboxes = document.querySelectorAll('.row-select');
    
    if (!selectAllCheckbox) return;

    selectAllCheckbox.addEventListener('change', function() {
        rowCheckboxes.forEach(checkbox => {
            checkbox.checked = this.checked;
        });
        updateBulkActions();
    });

    rowCheckboxes.forEach(checkbox => {
        checkbox.addEventListener('change', function() {
            updateSelectAllState();
            updateBulkActions();
        });
    });
}

function updateSelectAllState() {
    const selectAllCheckbox = document.getElementById('select-all');
    const rowCheckboxes = document.querySelectorAll('.row-select');
    const checkedCount = document.querySelectorAll('.row-select:checked').length;
    
    selectAllCheckbox.checked = checkedCount === rowCheckboxes.length && checkedCount > 0;
    selectAllCheckbox.indeterminate = checkedCount > 0 && checkedCount < rowCheckboxes.length;
}

function initializeBulkActions() {
    updateBulkActions();
}

function updateBulkActions() {
    const bulkActionsDiv = document.getElementById('bulk-actions');
    const selectedCountSpan = document.getElementById('selected-count');
    const checkedBoxes = document.querySelectorAll('.row-select:checked');
    
    if (!bulkActionsDiv || !selectedCountSpan) return;
    
    selectedCountSpan.textContent = checkedBoxes.length;
    
    if (checkedBoxes.length > 0) {
        bulkActionsDiv.style.display = 'block';
    } else {
        bulkActionsDiv.style.display = 'none';
    }
}

function initializeSearch() {
    const searchInput = document.getElementById('search');
    if (!searchInput) return;
    
    let searchTimeout;
    searchInput.addEventListener('input', function() {
        clearTimeout(searchTimeout);
        searchTimeout = setTimeout(() => {
            this.form.submit();
        }, 500);
    });
}

function initializeModals() {
    // Modal functionality
    window.showModal = function(modalId) {
        const modal = document.getElementById(modalId);
        if (modal) {
            modal.style.display = 'flex';
            document.body.style.overflow = 'hidden';
        }
    };
    
    window.closeModal = function(modalId) {
        const modal = document.getElementById(modalId);
        if (modal) {
            modal.style.display = 'none';
            document.body.style.overflow = '';
        }
    };
}

// Delete confirmation
let deleteId = null;

function confirmDelete(id, name) {
    deleteId = id;
    document.getElementById('deleteItemName').textContent = name;
    showModal('deleteModal');
}

document.getElementById('confirmDeleteBtn')?.addEventListener('click', function() {
    if (deleteId) {
        performDelete(deleteId);
    }
});

function performDelete(id) {
    // Show loading state
    this.disabled = true;
    this.textContent = 'Deleting...';
    
    fetch(`${window.location.pathname}/${id}`, {
        method: 'DELETE',
        headers: {
            'X-CSRF-TOKEN': document.querySelector('meta[name="csrf-token"]').content,
            'Accept': 'application/json',
            'Content-Type': 'application/json'
        }
    })
    .then(response => response.json())
    .then(data => {
        closeModal('deleteModal');
        
        if (data.success) {
            showAlert('success', data.message);
            setTimeout(() => location.reload(), 1000);
        } else {
            showAlert('error', data.message || 'Error deleting item');
        }
    })
    .catch(error => {
        closeModal('deleteModal');
        showAlert('error', 'An error occurred while deleting the item');
        console.error('Error:', error);
    });
}

function showAlert(type, message) {
    const alertDiv = document.createElement('div');
    alertDiv.className = `alert alert--${type}`;
    alertDiv.innerHTML = `
        <span class="alert-icon">${type === 'success' ? '✓' : '⚠'}</span>
        ${message}
        <button type="button" class="alert-close" onclick="this.parentElement.remove()">×</button>
    `;
    
    const container = document.querySelector('.container');
    container.insertBefore(alertDiv, container.firstChild);
    
    // Auto-dismiss after 5 seconds
    setTimeout(() => {
        if (alertDiv.parentNode) {
            alertDiv.remove();
        }
    }, 5000);
}
```

### 1. **Full Control**
- Complete control over styling and behavior
- No framework-specific limitations
- Easy to customize for brand requirements

### 2. **Performance**
- Smaller CSS bundle (only what you need)
- No unused framework code
- Optimized for your specific use cases

### 3. **Maintainability**
- Clear, semantic HTML structure
- Organized CSS architecture
- Easy to understand and modify

### 4. **Flexibility**
- Can integrate with any design system
- Easy to add new components
- Framework-agnostic approach

### 5. **Future-Proof**
- Not tied to external framework updates
- Modern CSS features (Grid, Flexbox, Custom Properties)
- Easy migration path
